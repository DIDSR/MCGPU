00503 ////////////////////////////////////////////////////////////////////////////////
00504 //!  Tally a radiographic projection image.
00505 //!  This function is called whenever a particle escapes the voxelized volume.
00506 //!  The code checks if the particle would arrive at the detector if it kept
00507 //!  moving in a straight line after exiting the voxels (assuming vacuum enclosure).
00508 //!  An ideal image formation model is implemented: each pixel counts the total energy
00509 //!  of the x rays that enter the pixel (100% detection efficiency for any energy).
00510 //!  The image due to primaries and different kinds of scatter is tallied separately.
00511 //!
00512 //!  In the GPU, and atomicAdd() function is used to make sure that multiple threads do
00513 //!  not update the same pixel at the same time, which would result in a lose of information.
00514 //!  Since the atomicAdd function is only available for 'unsigned long long int' data,
00515 //!  the float pixel values are scaled by a factor "SCALE_eV" defined in the header file
00516 //!  (eg, #define SCALE_eV 10000.0f) and stored as unsigned long long integers in main
00517 //!  memory.
00518 //!
00519 //!  WARNING! If the total tallied signal (for all particles) is larger than "1.8e19/SCALE_eV",
00520 //!    there will be a bit overflow and the value will be reset to 0 giving bogus results.
00521 //!
00522 //!  WARNING! The detector plane should be located outside the voxels bounding box. However, since
00523 //!    the particles are moved outside the bbox in the last step, they could cross the detector 
00524 //!    plane anyway. If the particles are less than 2.0 cm behind the detector, they are moved 
00525 //!    back and detected. Therefore the detector can be a few cm inside the bbox and still work.
00526 //!    If the Woodcock mean free path is larger than the distance from the bbox to the detector, 
00527 //!    we may lose some particles behind the detector!
00528 //!
00529 //!
00530 //!       @param[in] energy   X-ray energy
00531 //!       @param[in] position   Particle position
00532 //!       @param[in] direction   Particle direction (cosine vectors)
00533 //!       @param[in] scatter_state  Flag marking primaries, single Compton, single Rayleigh or multiple scattered radiation
00534 //!       @param[out] image   Integer array containing the image, ie, the pixel values (in tenths of meV)
00535 ////////////////////////////////////////////////////////////////////////////////
00536 #ifdef USING_CUDA
00537 __device__
00538 #endif

00540 {

00544   {
00545     
00546     // *** Skip particles not moving towards the detector. 
00547     //       (a) Skip particles that were deflected more than 90 deg from the original source direction (backscatter).
00548     //       (b) Skip particles located more than 10 cm behind the detector 
00549     //       (c) Skip particles for which the direction to the detector is way bigger than SDD (likely to intersect the plane outside the pixel region).
00550                   // !!DeBuG!! NOTE: This may give problems for big detectors very close to the source
00551                   
00552     //      !!DeBuG!! !!WARNING!! Particles located after the detector will be moved back to the surface of the detector, but 10 cm maximum!!
00553     //      !!DeBuG!!     In this way the detector can intersect the voxels bbox or be located right on the surface of the bbox: the particles will be 
00554     //      !!DeBuG!!     transported across the detector and until a little after the end of the bbox in the last step, but then moved back.
00555     //      !!DeBuG!!     This algorithm will give correct results ONLY when the detector intersects just slightly the air space around the phantom,
00556     //      !!DeBuG!!     so that the interactions after the detector are not significant (this happens sometimes using oblique beams).
00557     //      !!DeBuG!!     I could remove particles after the detector using "if (dist_detector<0.0f) return;".
00558 
00559 
00560     //  (a) Calculate the angle between the particle and the initial direction (dot product): reject particle if cos_angle < cos(89)==0 (angle>89deg):
00561     //      [Extra parenthesis are coded to suggest to the compiler the use of intrinsic multiply-add operations].
00562

00564                               (direction->y * source_data_CONST.direction[*num_p].y +

00566     if (cos_angle < 0.025f)
00567       return;  // Reject particle: Angle larger than 89 deg --> particle moving parallel to the detector or backwards towards the source!

00571                      (source_data_CONST.direction[*num_p].y * (detector_center_SHARED->y - position->y) +
00572                      (source_data_CONST.direction[*num_p].z * (detector_center_SHARED->z - position->z))) ) / cos_angle;
00573 
00574                         
00575     if (dist_detector < -10.0f)   // !!DeBuG!! Is 10 cm enough or too much? Should I use 0? or allow any distance?
00576       return;  // !!DeBuG!! Reject particles located more than 10 cm behind the detector. 10 cm was selected arbitrarily. Woodcock MFP for x-rays in bone: MFP 200 keV photons in bone ==> 4 cm.
00577       
00578     if (fabsf(dist_detector)>(2.1f*detector_data_CONST.sdd))          
00579       return;  // Reject particle: distance to the detector plane too large, the particle is likely to travel almost parallel to the detector and will not be detected.
00580 
00581             
00582     // *** Translate the particle to the detector plane (we assume the detector is completely absorbent: 100% detection efficiency):
00583     position->x = position->x + dist_detector * direction->x;
00584     position->y = position->y + dist_detector * direction->y;
00585     position->z = position->z + dist_detector * direction->z;
00586 
00587     // *** Rotate the particle position vector to the default reference system where the detector is perpendicular to the +Y axis, then find out if the particle is located inside a pixel:
00588     #ifdef USING_CUDA
00589       rotated_position = detector_data_CONST.rot_inv[*num_p][0]*position->x + detector_data_CONST.rot_inv[*num_p][1]*position->y + detector_data_CONST.rot_inv[*num_p][2]*position->z;  // X coordinate
00590       int pixel_coord_x = __float2int_rd((rotated_position - detector_data_CONST.corner_min_rotated_to_Y[*num_p].x) * detector_data_CONST.inv_pixel_size_X);    // Using CUDA intrinsic function to convert float to integer rounding down (towards minus infinite)

00592       {
00593         rotated_position = detector_data_CONST.rot_inv[*num_p][6]*position->x + detector_data_CONST.rot_inv[*num_p][7]*position->y + detector_data_CONST.rot_inv[*num_p][8]*position->z;  // Z coordinate

00595         if ((pixel_coord_z>-1)&&(pixel_coord_z<detector_data_CONST.num_pixels.y))
00596         {
00597           // -- Particle enters the detector! Tally the particle energy in the corresponding pixel (in tenths of meV):
00598           //    Using a CUDA atomic function (not available for global floats yet) to read and increase the pixel value in a single instruction, blocking interferences from other threads.
00599           //    The offset for the primaries or scatter images are calculated considering that:
00600           //      scatter_state=0 for non-scattered, =1 for Compton, =2 for Rayleigh, and =3 for multiple scatter.

00606         }

00608     #else
00609       // CPU version (not using CUDA intrinsics: atomicAdd, fast type casting)
00610       rotated_position = detector_data_CONST.rot_inv[*num_p][0]*position->x + detector_data_CONST.rot_inv[*num_p][1]*position->y + detector_data_CONST.rot_inv[*num_p][2]*position->z;  // X coordinate
00611       int pixel_coord_x = (int)(floor((rotated_position - detector_data_CONST.corner_min_rotated_to_Y[*num_p].x)*detector_data_CONST.inv_pixel_size_X) + 0.00001f);   // Adding EPS to have correct truncation to int
00612 
00613       if ((pixel_coord_x>-1)&&(pixel_coord_x<detector_data_CONST.num_pixels.x))

00619              += (unsigned long long int)((*energy)*SCALE_eV + 0.5f);  // Tally the particle energy in the pixel. This instruction is not thread-safe, but it is ok in sequential code.          
00620       }
00621     #endif
00622   }
00623   else  // (detector_data_CONST.rotation_flag != 1) -->  Initial source direction is (0,1,0): pixel number and distance can be found easily
00624   {

00626   
00627     if (direction->y < 0.0001f)
00628       return;  // *** Reject particles not moving towards the detector plane at +Y.
00629 
00630     dist_detector = (detector_center_SHARED->y - position->y)/(direction->y);  // Distance to the intersection with the detector at +Y.
00631 
00632     if (dist_detector>(2.1f*detector_data_CONST.sdd))
00633       return;  // *** Eliminate particles that intersect the detector plane too far away
00634      
00635     #ifdef USING_CUDA
00636     int pixel_coord_x = __float2int_rd((position->x + dist_detector*direction->x - detector_data_CONST.corner_min_rotated_to_Y[*num_p].x)*detector_data_CONST.inv_pixel_size_X);
00637     if ((pixel_coord_x>-1)&&(pixel_coord_x<detector_data_CONST.num_pixels.x))
00638     {
00639       int pixel_coord_z = __float2int_rd((position->z + dist_detector*direction->z - detector_data_CONST.corner_min_rotated_to_Y[*num_p].z)*detector_data_CONST.inv_pixel_size_Z);

00641         atomicAdd( ( image +                                                                // Pointer to beginning of image array
00642                      (int)(*scatter_state) * detector_data_CONST.total_num_pixels +         // Offset to corresponding scatter image
00643                      (pixel_coord_x + pixel_coord_z*(detector_data_CONST.num_pixels.x)) ),  // Offset to the corresponding pixel
00644                    __float2ull_rn((*energy)*SCALE_eV) );    // Energy arriving at the pixel, scaled by the factor SCALE_eV and rounded.
00645     }
00646     #else
00647 
00648     // --Calculate the pixel the xray enters, truncating towards minus infinite and making sure the conversion to int is safe:
00649     float pixel_coord_x = floor((position->x + dist_detector*direction->x - detector_data_CONST.corner_min_rotated_to_Y[*num_p].x)*detector_data_CONST.inv_pixel_size_X);
00650 
00651     if ( (pixel_coord_x>-0.5f) && (pixel_coord_x<(detector_data_CONST.num_pixels.x-0.5f)) )
00652     {
00653       float pixel_coord_z = floor((position->z + dist_detector*direction->z - detector_data_CONST.corner_min_rotated_to_Y[*num_p].z)*detector_data_CONST.inv_pixel_size_Z);
00654       if ( (pixel_coord_z>-0.5f) && (pixel_coord_z<(detector_data_CONST.num_pixels.y-0.5f)) )
00655         image[((int)(*scatter_state))*detector_data_CONST.total_num_pixels + (int)(pixel_coord_x+0.001f)+ (int)(pixel_coord_z+0.001f)*detector_data_CONST.num_pixels.x] += (unsigned long long int)((*energy)*SCALE_eV + 0.5f);
00656     }
00657     #endif
00658   }

00663 ////////////////////////////////////////////////////////////////////////////////
00664 //!  Source that creates primary x rays, according to the defined source model.
00665 //!  The particles are automatically moved to the surface of the voxel bounding box,
00666 //!  to start the tracking inside a real material. If the sampled particle do not
00667 //!  enter the voxels, it is init in the focal spot and the main program will check
00668 //!  if it arrives at the detector or not.
00669 //!
00670 //!       @param[in] source_data   Structure describing the source.
00671 //!       @param[out] position   Initial particle position (particle transported inside the voxel bbox).
00672 //!       @param[out] direction   Sampled particle direction (cosine vectors).
00673 //!       @param[out] energy   Sampled energy of the new x ray.
00674 //!       @param[in] seed   Current seed of the random number generator, requiered to sample the movement direction.
00675 //!       @param[out] absvox   Set to <0 if primary particle will not cross the voxels, not changed otherwise (>0).
00676 ////////////////////////////////////////////////////////////////////////////////
00677 #ifdef USING_CUDA
00678 __device__
00679 #endif
00680 inline void source(int* num_p, float3* position, float3* direction, float* energy, int2* seed, int* absvox)
00681 {
00682   if (1 == source_data_CONST.num_bins_espc)
00683   {
00684   // *** Assign initial energy for a monoenergetic beam:
00685       *energy = source_data_CONST.mean_energy;
00686   }
00687   else
00688   {
00689   // *** Sample the initial x-ray energy following the input energy spectrum using the Walker aliasing algorithm from PENELOPE:
00690           //The following code is equivalent to calling the function "seeki_walker": int sampled_bin = seeki_walker(source_data_CONST.espc_cutoff, source_data_CONST.espc_alias, ranecu(seed), source_data_CONST.num_bins_espc);      
00691     int sampled_bin;
00692     float RN = ranecu(seed) * source_data_CONST.num_bins_espc;    // Find initial interval (array starting at 0):   
00693     #ifdef USING_CUDA
00694       int int_part = __float2int_rd(RN);                          //   -- Integer part (round down)
00695     #else
00696       int int_part = (int)(RN);
00697     #endif
00698     float fraction_part = RN - ((float)int_part);                 //   -- Fractional part
00699     if (fraction_part < source_data_CONST.espc_cutoff[int_part])  // Check if we are in the aliased part
00700       sampled_bin = int_part;                                     // Below the cutoff: return current value
00701     else
00702       sampled_bin = (int)source_data_CONST.espc_alias[int_part];  // Above the cutoff: return alias
00703     
00704     // Linear interpolation of the final energy within the sampled energy bin
00705     *energy = source_data_CONST.espc[sampled_bin] + ranecu(seed) * (source_data_CONST.espc[sampled_bin+1] - source_data_CONST.espc[sampled_bin]);   
00706   }
00707     
00708 // printf(" !!Walker!! Sampled energy [bin=%d] = %f keV\n", sampled_bin, 0.001f*(*energy));  //!!DeBuG!! Verbose Test sampling

00712  
00713    // *** Sample the initial direction:
00714

00716   {
00717     //     Using the algorithm used in PENMAIN.f, from penelope 2008 (by F. Salvat).
00718     direction->z = source_data_CONST.cos_theta_low + ranecu(seed)*source_data_CONST.D_cos_theta;     // direction->z = w = cos(theta_sampled)

00723     #ifdef USING_CUDA

00728     #endif

00776   {

00778     //    An EPSILON distance is added to make sure the particles will be clearly inside the bbox, not on the surface.
00779     //    Currently the source can not be located inside the bbox (particles will be moved to the edge anyway)!
00780     //    The back lower vertex of the voxel bounding box is always located at the origin: (x0,y0,z0)=(0,0,0).

00801     }

00803       dist_y = INF;   // Particle moving parallel to the plane: considering no interaction possible (dist = INFINITE)

00809       if (source_data_CONST.position[*num_p].x > 0.0f)
00810         dist_x = INF;  // No intersection with this plane possible: dist_x would be negative
00811       else  
00812         dist_x = EPS_SOURCE + (-source_data_CONST.position[*num_p].x)/(direction->x);    // dist_x > 0 for sure in this case
00813     }

00820         dist_x = EPS_SOURCE + (voxel_data_CONST.size_bbox.x - source_data_CONST.position[*num_p].x)/(direction->x);    // dist_x > 0 for sure in this case

00834     else if ((direction->z) < NEG_EPS_SOURCE)

00841     }

00847     //  If the particle doesn't intersect the bbox, the focal point position is returned and the main program will take care of the particle outside the voxels.

00851       {

00853           break;   // Position found inside the voxel bbox, accept it.
00854         else
00855           dist_y = INF;
00856       }
00857       else if ((dist_x < INF_minus1) && (dist_x <  dist_z))   // X is the shortest distance
00858       {
00859         if (set_position(num_p, &dist_x, position, direction)==1)
00860           break;
00861         else
00862           dist_x = INF;
00863       }
00864       else if (dist_z < INF_minus1)    // Z is the shortest distance
00865       {
00866         if (set_position(num_p, &dist_z, position, direction)==1)
00867           break;
00868         else
00869           dist_z = INF;
00870       }
00871       else
00872       {
00873         // Particle not pointing to the voxel bbox: return focal spot position and absvox<0.
00874         (*absvox) = -111;

00876         position->y = source_data_CONST.position[*num_p].y;

00880     }

00920 //!  Upper limit of the number of random values sampled in a single track.
00921 #define  LEAP_DISTANCE     256
00922 //!  Multipliers and moduli for the two MLCG in RANECU.
00923 #define  a1_RANECU       40014
00924 #define  m1_RANECU  2147483563
00925 #define  a2_RANECU       40692
00926 #define  m2_RANECU  2147483399
00927 ////////////////////////////////////////////////////////////////////////////////
00928 //! Initialize the pseudo-random number generator (PRNG) RANECU to a position
00929 //! far away from the previous history (leap frog technique).
00930 //!
00931 //! Each calculated seed initiates a consecutive and disjoint sequence of
00932 //! pseudo-random numbers with length LEAP_DISTANCE, that can be used to
00933 //! in a parallel simulation (Sequence Splitting parallelization method).
00934 //! The basic equation behind the algorithm is:
00935 //!    S(i+j) = (a**j * S(i)) MOD m = [(a**j MOD m)*S(i)] MOD m  ,
00936 //! which is described in:
00937 //!   P L'Ecuyer, Commun. ACM 31 (1988) p.742
00938 //!
00939 //! This function has been adapted from "seedsMLCG.f", see:
00940 //!   A Badal and J Sempau, Computer Physics Communications 175 (2006) p. 440-450
00941 //!
00942 //!       @param[in] history   Particle bach number.
00943 //!       @param[in] seed_input   Initial PRNG seed input (used to initiate both MLCGs in RANECU).
00944 //!       @param[out] seed   Initial PRNG seeds for the present history.
00945 //!
00946 ////////////////////////////////////////////////////////////////////////////////
00947 #ifdef USING_CUDA
00948 __device__
00949 #endif
00950 inline void init_PRNG(int history_batch, int histories_per_thread, int seed_input, int2* seed)
00951 {
00952   // -- Move the RANECU generator to a unique position for the current batch of histories:
00953   //    I have to use an "unsigned long long int" value to represent all the simulated histories in all previous batches
00954   //    The maximum unsigned long long int value is ~1.8e19: if history >1.8e16 and LEAP_DISTANCE==1000, 'leap' will overflow.
00955   // **** 1st MLCG:
00956   unsigned long long int leap = ((unsigned long long int)(history_batch+1))*(histories_per_thread*LEAP_DISTANCE);
00957   int y = 1;
00958   int z = a1_RANECU;
00959   // -- Calculate the modulo power '(a^leap)MOD(m)' using a divide-and-conquer algorithm adapted to modulo arithmetic
00960   for(;;)
00961   {
00962     // (A2) Halve n, and store the integer part and the residue
00963     if (0!=(leap&01))  // (bit-wise operation for MOD(leap,2), or leap%2 ==> proceed if leap is an odd number)  Equivalent: t=(short)(leap%2);
00964     {
00965       leap >>= 1;     // Halve n moving the bits 1 position right. Equivalent to:  leap=(leap/2);  
00966       y = abMODm(m1_RANECU,z,y);      // (A3) Multiply y by z:  y = [z*y] MOD m
00967       if (0==leap) break;         // (A4) leap==0? ==> finish
00968     }
00969     else           // (leap is even)
00970     {
00971       leap>>= 1;     // Halve leap moving the bits 1 position right. Equivalent to:  leap=(leap/2);
00972     }
00973     z = abMODm(m1_RANECU,z,z);        // (A5) Square z:  z = [z*z] MOD m
00974   }
00975   // AjMODm1 = y;                 // Exponentiation finished:  AjMODm = expMOD = y = a^j
00976 
00977   // -- Compute and display the seeds S(i+j), from the present seed S(i), using the previously calculated value of (a^j)MOD(m):
00978   //         S(i+j) = [(a**j MOD m)*S(i)] MOD m
00979   //         S_i = abMODm(m,S_i,AjMODm)
00980   seed->x = abMODm(m1_RANECU, seed_input, y);     // Using the input seed as the starting seed
00981 
00982   // **** 2nd MLCG (repeating the previous calculation for the 2nd MLCG parameters):
00983   leap = ((unsigned long long int)(history_batch+1))*(histories_per_thread*LEAP_DISTANCE);
00984   y = 1;
00985   z = a2_RANECU;
00986   for(;;)
00987   {
00988     // (A2) Halve n, and store the integer part and the residue
00989     if (0!=(leap&01))  // (bit-wise operation for MOD(leap,2), or leap%2 ==> proceed if leap is an odd number)  Equivalent: t=(short)(leap%2);
00990     {
00991       leap >>= 1;     // Halve n moving the bits 1 position right. Equivalent to:  leap=(leap/2);
00992       y = abMODm(m2_RANECU,z,y);      // (A3) Multiply y by z:  y = [z*y] MOD m
00993       if (0==leap) break;         // (A4) leap==0? ==> finish
00994     }
00995     else           // (leap is even)
00996     {
00997       leap>>= 1;     // Halve leap moving the bits 1 position right. Equivalent to:  leap=(leap/2);
00998     }
00999     z = abMODm(m2_RANECU,z,z);        // (A5) Square z:  z = [z*z] MOD m
01000   }
01001   // AjMODm2 = y;
01002   seed->y = abMODm(m2_RANECU, seed_input, y);     // Using the input seed as the starting seed
01003 }

01007 /////////////////////////////////////////////////////////////////////
01008 //!  Calculate "(a1*a2) MOD m" with 32-bit integers and avoiding
01009 //!  the possible overflow, using the Russian Peasant approach
01010 //!  modulo m and the approximate factoring method, as described
01011 //!  in:  L'Ecuyer and Cote, ACM Trans. Math. Soft. 17 (1991).
01012 //!
01013 //!  This function has been adapted from "seedsMLCG.f", see: 
01014 //!  Badal and Sempau, Computer Physics Communications 175 (2006)
01015 //!
01016 //!       @param[in] m,a,s  MLCG parameters
01017 //!       @return   (a1*a2) MOD m   
01018 //
01019 //    Input:          0 < a1 < m                                  
01020 //                    0 < a2 < m                                  
01021 //
01022 //    Return value:  (a1*a2) MOD m                                
01023 //
01024 /////////////////////////////////////////////////////////////////////
01025 #ifdef USING_CUDA
01026 __device__ __host__    // !!DeBuG!! MPI: Function will be callable from host and also from device
01027 #endif
01028 inline int abMODm(int m, int a, int s)
01029 {
01030   // CAUTION: the input parameters are modified in the function but should not be returned to the calling function! (pass by value!)
01031   int q, k;
01032   int p = -m;            // p is always negative to avoid overflow when adding
01033 
01034   // ** Apply the Russian peasant method until "a =< 32768":
01035   while (a>32768)        // We assume '32' bit integers (4 bytes): 2^(('32'-2)/2) = 32768
01036   {
01037     if (0!=(a&1))        // Store 's' when 'a' is odd     Equivalent code:   if (1==(a%2))
01038     {
01039       p += s;
01040       if (p>0) p -= m;
01041     }
01042     a >>= 1;             // Half a (move bits 1 position right)   Equivalent code: a = a/2;
01043     s = (s-m) + s;       // Double s (MOD m)
01044     if (s<0) s += m;     // (s is always positive)
01045   }
01046 
01047   // ** Employ the approximate factoring method (a is small enough to avoid overflow):
01048   q = (int) m / a;
01049   k = (int) s / q;
01050   s = a*(s-k*q)-k*(m-q*a);
01051   while (s<0)
01052     s += m;
01053 
01054   // ** Compute the final result:
01055   p += s;
01056   if (p<0) p += m;
01057 
01058   return p;
01059 }

01178 //////////////////////////////////////////////////////////////////////
01179 //!   Rotates a vector; the rotation is specified by giving
01180 //!   the polar and azimuthal angles in the "self-frame", as
01181 //!   determined by the vector to be rotated.
01182 //!   This function is a literal translation from Fortran to C of
01183 //!   PENELOPE (v. 2006) subroutine "DIRECT".
01184 //!
01185 //!    @param[in,out]  (u,v,w)  input vector (=d) in the lab. frame; returns the rotated vector components in the lab. frame
01186 //!    @param[in]  costh  cos(theta), angle between d before and after turn
01187 //!    @param[in]  phi  azimuthal angle (rad) turned by d in its self-frame
01188 //
01189 //    Output:
01190 //      (u,v,w) -> rotated vector components in the lab. frame
01191 //
01192 //    Comments:
01193 //      -> (u,v,w) should have norm=1 on input; if not, it is
01194 //         renormalized on output, provided norm>0.
01195 //      -> The algorithm is based on considering the turned vector
01196 //         d' expressed in the self-frame S',
01197 //           d' = (sin(th)cos(ph), sin(th)sin(ph), cos(th))
01198 //         and then apply a change of frame from S' to the lab
01199 //         frame. S' is defined as having its z' axis coincident
01200 //         with d, its y' axis perpendicular to z and z' and its
01201 //         x' axis equal to y'*z'. The matrix of the change is then
01202 //                   / uv/rho    -v/rho    u \
01203 //          S ->lab: | vw/rho     u/rho    v |  , rho=(u^2+v^2)^0.5
01204 //                   \ -rho       0        w /
01205 //      -> When rho=0 (w=1 or -1) z and z' are parallel and the y'
01206 //         axis cannot be defined in this way. Instead y' is set to
01207 //         y and therefore either x'=x (if w=1) or x'=-x (w=-1)
01208 //////////////////////////////////////////////////////////////////////
01209 #ifdef USING_CUDA
01210 __device__
01211 #endif
01212 inline void rotate_double(float3* direction, double costh, double phi)   // !!DeBuG!! The direction vector is single precision but the rotation is performed in doule precision for increased accuracy.
01213 {
01214   double DXY, NORM, cosphi, sinphi, SDT;
01215   DXY = direction->x*direction->x + direction->y*direction->y;
01216   
01217 #ifdef USING_CUDA
01218   sincos(phi, &sinphi,&cosphi);   // Calculate the SIN and COS at the same time.
01219 #else
01220   sinphi = sin(phi);   // Some CPU compilers will be able to use "sincos", but let's be safe. (!!DeBuG!!)
01221   cosphi = cos(phi);
01222 #endif   
01223 
01224   // ****  Ensure normalisation
01225   NORM = DXY + direction->z*direction->z;     // !!DeBuG!! Check if it is really necessary to renormalize in a real simulation!!
01226   if (fabs(NORM-1.0)>1.0e-14)
01227   {
01228     NORM = 1.0/sqrt(NORM);
01229     direction->x = NORM*direction->x;
01230     direction->y = NORM*direction->y;
01231     direction->z = NORM*direction->z;
01232     DXY = direction->x*direction->x + direction->y*direction->y;
01233   }
01234   if (DXY>1.0e-28)
01235   {
01236     SDT = sqrt((1.0-costh*costh)/DXY);
01237     float direction_x_in = direction->x;
01238     direction->x = direction->x*costh + SDT*(direction_x_in*direction->z*cosphi-direction->y*sinphi);
01239     direction->y = direction->y*costh+SDT*(direction->y*direction->z*cosphi+direction_x_in*sinphi);
01240     direction->z = direction->z*costh-DXY*SDT*cosphi;
01241   }
01242   else
01243   {
01244     SDT = sqrt(1.0-costh*costh);
01245     direction->y = SDT*sinphi;
01246     if (direction->z>0.0)
01247     {
01248       direction->x = SDT*cosphi;
01249       direction->z = costh;
01250     }
01251     else
01252     {
01253       direction->x =-SDT*cosphi;
01254       direction->z =-costh;
01255     }
01256   }
01257 }
